Maven Plugin Tool for Annotations

The Maven Plugin Tool for Annotations is the java-annotations implementation of maven-plugin-tools-api to extract descriptors from plugins written in Java with Maven Plugin Tools Java Annotations.

Supported Annotations

  1. import org.apache.maven.execution.MavenSession;
  2. import org.apache.maven.plugin.AbstractMojo;
  3. import org.apache.maven.plugin.MojoExecution;
  4. import org.apache.maven.plugin.descriptor.PluginDescriptor;
  5. import org.apache.maven.plugins.annotations.Component;
  6. import org.apache.maven.plugins.annotations.Execute;
  7. import org.apache.maven.plugins.annotations.InstantiationStrategy;
  8. import org.apache.maven.plugins.annotations.LifecyclePhase;
  9. import org.apache.maven.plugins.annotations.Mojo;
  10. import org.apache.maven.plugins.annotations.Parameter;
  11. import org.apache.maven.plugins.annotations.ResolutionScope;
  12. import org.apache.maven.project.MavenProject;
  13. import org.apache.maven.settings.Settings;
  14.  
  15. /**
  16.  * Mojo Description. @Mojo( name = "<goal-name>" ) is the minimal required annotation.
  17.  * @since <since-text>
  18.  * @deprecated <deprecated-text>
  19.  */
  20. @Mojo( name = "<goal-name>",
  21.        aggregator = <false|true>, 
  22.        configurator = "<role hint>",
  23.        executionStrategy = "<once-per-session|always>",
  24.        inheritByDefault = <true|false>,
  25.        instantiationStrategy = InstantiationStrategy.<strategy>,
  26.        defaultPhase = LifecyclePhase.<phase>,
  27.        requiresDependencyResolution = ResolutionScope.<scope>,
  28.        requiresDependencyCollection = ResolutionScope.<scope>, // (since Maven 3.0)
  29.        requiresDirectInvocation = <false|true>,
  30.        requiresOnline = <false|true>,
  31.        requiresProject = <true|false>,
  32.        requiresReports = <false|true>, // (unsupported since Maven 3.0)
  33.        threadSafe = <false|true> ) // (since Maven 3.0)
  34. @Execute( goal = "<goal-name>",
  35.           phase = LifecyclePhase.<phase>,
  36.           lifecycle = "<lifecycle-id>" )
  37. public class MyMojo
  38.     extends AbstractMojo
  39. {
  40.     /**
  41.      * @since <since-text>
  42.      * @deprecated <deprecated-text>
  43.      */
  44.     @Parameter( name = "parameter",
  45.                 alias = "myAlias",
  46.                 property = "a.property",
  47.                 defaultValue = "an expression, possibly with ${variables}",
  48.                 readonly = <false|true>,
  49.                 required = <false|true> )
  50.     private String parameter;
  51.  
  52.     @Component( role = MyComponentExtension.class,
  53.                 hint = "..." )
  54.     private MyComponent component;
  55.  
  56.     // sample objects taken from Maven API through PluginParameterExpressionEvaluator
  57.  
  58.     @Parameter( defaultValue = "${session}", readonly = true )
  59.     private MavenSession session;
  60.  
  61.     @Parameter( defaultValue = "${project}", readonly = true )
  62.     private MavenProject project;
  63.  
  64.     @Parameter( defaultValue = "${mojoExecution}", readonly = true )
  65.     private MojoExecution mojo;
  66.  
  67.     @Parameter( defaultValue = "${plugin}", readonly = true ) // Maven 3 only
  68.     private PluginDescriptor plugin;
  69.  
  70.     @Parameter( defaultValue = "${settings}", readonly = true )
  71.     private Settings settings;
  72.  
  73.     @Parameter( defaultValue = "${project.basedir}", readonly = true )
  74.     private File basedir;
  75.  
  76.     @Parameter( defaultValue = "${project.build.directory}", readonly = true )
  77.     private File target;
  78.  
  79.     public void execute()
  80.     {
  81.         ...
  82.     }
  83. }
